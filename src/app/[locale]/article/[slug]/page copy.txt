import { notFound } from 'next/navigation';
import { Metadata } from 'next';

type Post = {
  id: number;
  title: string;
  excerpt: string;
  content: string;
  slug: string;
  link: string;
  featured_image: string;
  date_published: string;
  author: string;
  categories: string[];
  categoriesdetails: {
    name: string;
    slug: string;
    id: number;
  }[];
};

// ✅ Meta avec Promise pour params
export async function generateMetadata({
  params,
}: {
  params: Promise<{ slug: string }>;
}): Promise<Metadata> {
  const { slug } = await params;

  try {
    const res = await fetch(`https://siteco.cynomedia-africa.com/wp-json/custom/v1/article/${slug}`, 
      {next: { revalidate: 60 },}
    );

    if (!res.ok) return {};

    const post: Post = await res.json();
    return {
      title: post.title,
      description: post.excerpt,
      openGraph: {
        title: post.title,
        description: post.excerpt,
        images: [post.featured_image],
        url: post.link,
        type: 'article',
      },
      twitter: {
        card: 'summary_large_image',
        title: post.title,
        description: post.excerpt,
        images: [post.featured_image],
      },
    };
  } catch {
    return {};
  }
}





// ✅ Page principale sans Promise
export default async function PostPage({
  params,
}: {
  params: Promise<{ slug: string }>;
}) {
  const { slug } = await params;

  try {
    const res = await fetch(
      `https://siteco.cynomedia-africa.com/wp-json/custom/v1/article/${slug}`,
      { next: { revalidate: 60 } }
    );

    if (!res.ok) return notFound();

    const post: Post = await res.json();
    if (!post || !post.id) return notFound();

    return (
      <main className="p-4">
        <h1 className="text-2xl font-bold">{post.title}</h1>
        <img src={post.featured_image} alt={post.title} className="my-4" />
        <div
          dangerouslySetInnerHTML={{ __html: post.content }}
          className="prose"
        />
      </main>
    );
  } catch (error) {
    console.error('Error fetching post:', error);
    return notFound();
  }
}
