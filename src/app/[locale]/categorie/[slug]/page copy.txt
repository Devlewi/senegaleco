// ✅ Meta avec Promise pour params
export async function generateMetadata({ params }: Props) {
  const { locale, slug } = await params;

  try {
    // Récupérer les publications de la catégorie spécifique
    const res = await fetch(
      `https://siteco.cynomedia-africa.com/wp-json/custom-api/publications/slug-categorie/${slug}/`
    );

    if (!res.ok) return {};

    const data = await res.json();
    const firstPost = data.articles[0];

    return {
      title: firstPost.title,
      description: firstPost.excerpt,
      openGraph: {
        title: firstPost.title,
        description: firstPost.excerpt,
        images: [firstPost.featured_image],
        url: firstPost.link,
        type: "article",
      },
      twitter: {
        card: "summary_large_image",
        title: firstPost.title,
        description: firstPost.excerpt,
        images: [firstPost.featured_image],
      },
    };
  } catch {
    return {};
  }
}
