"use client"; // Indique que le code est destiné au rendu côté client

import LanguageSelector from "@/components/LanguageSelector";
import { useTranslations } from "next-intl";
import Link from "next/link";
import { usePathname } from "next/navigation";
import { useState, useEffect, useRef } from "react";
import {
  FaSearch,
  FaTimes,
  FaFacebookF,
  FaLinkedinIn,
  FaTwitter,
  FaYoutube,
} from "react-icons/fa"; // Icônes de réseaux sociaux
import { translatedSlugs } from "../lib/translatedSlugs";



export default function Navbar() {

  const [isMobileMenuOpen, setIsMobileMenuOpen] = useState(false);
  const [, setTokenStatus] = useState<string>("supprimé");
  const [scrolled, setScrolled] = useState(false); // État pour le scroll
  const menuRef = useRef<HTMLDivElement | null>(null);
  //const { t } = useTranslations("common"); // ou ton namespace i18n
  //const { locale } = useRouter(); // détecte la langue en cours

  //const menuRef = useRef(null); // Référence pour le menu mobile

  const toggleMobileMenu = () => {
    setIsMobileMenuOpen(!isMobileMenuOpen);
  };

  const t = useTranslations();

  // Récupérer la locale à partir de l'URL
  const pathname = usePathname();
  const locale = pathname.split("/")[1]; // Récupère la locale à partir du chemin de l'URL


  const menuKeys = [
    "propos",
    "societe",
    "metier",
    "projet",
    "actualite",
    "developpement_durable",
    "carriere",
    "publications"
  ];
  
  const navItems = menuKeys.map((key) => ({
    name: t(key),
    href: `/${locale}/${translatedSlugs[key]?.[locale] ?? key}`, // slug dynamique
  }));
  

  useEffect(() => {
    const checkTokenStatus = () => {
      const token = localStorage.getItem("token");
      setTokenStatus(token ? "existe" : "supprimé");
    };

    checkTokenStatus();

    const syncTokenState = (event: StorageEvent) => {
      if (event.key === "token") {
        checkTokenStatus();
      }
    };

    window.addEventListener("storage", syncTokenState);

    return () => {
      window.removeEventListener("storage", syncTokenState);
    };
  }, []);

  // Détection du scroll pour changer la couleur du fond de nav
  useEffect(() => {
    const handleScroll = () => {
      if (window.scrollY > 50) {
        setScrolled(true); // Lorsque l'utilisateur défile
      } else {
        setScrolled(false); // Sinon, on remet l'état à false
      }
    };

    window.addEventListener("scroll", handleScroll);

    return () => {
      window.removeEventListener("scroll", handleScroll);
    };
  }, []);

  
  
  // Fermer le menu lorsque l'utilisateur clique en dehors
  useEffect(() => {
    const handleClickOutside = (event: MouseEvent) => {
      if (menuRef.current && !menuRef.current.contains(event.target as Node)) {
        setIsMobileMenuOpen(false);
      }
    };

    document.addEventListener("mousedown", handleClickOutside);
    return () => {
      document.removeEventListener("mousedown", handleClickOutside);
    };
  }, []);

  return (
    <div>
      <nav
        className={`fixed top-0 left-0 w-full z-50 py-0 transition-all duration-300 ${
          scrolled ? "bg-[#061443]" : "bg-gradient-to-b from-[#031354]/90 to-transparent"
        }`}
      >
        <div className="w-[90%] mx-auto border-b-2 border-white py-4">
          {/* Icônes réseaux sociaux en haut à droite */}
          <div className="flex items-center justify-end gap-1 px-6 py-2">
            {/* Facebook */}
            <Link
              href="https://www.facebook.com"
              target="_blank"
              className="bg-white p-2 w-7 h-7 flex items-center justify-center text-[#031354] text-[30px] rounded-md"
              style={{ borderRadius: 2 }}
            >
              <FaFacebookF />
            </Link>

            {/* LinkedIn */}
            <Link
              href="https://www.linkedin.com"
              target="_blank"
              className="bg-white p-2 w-7 h-7 flex items-center justify-center text-[#031354] text-[30px] rounded-md"
              style={{ borderRadius: 2 }}
            >
              <FaLinkedinIn />
            </Link>

            {/* Twitter */}
            <Link
              href="https://www.twitter.com"
              target="_blank"
              className="bg-white p-2 w-7 h-7 flex items-center justify-center text-[#031354] text-[30px] rounded-md"
              style={{ borderRadius: 2 }}
            >
              <FaTwitter />
            </Link>

            {/* YouTube */}
            <Link
              href="https://www.youtube.com"
              target="_blank"
              className="bg-white p-2 w-7 h-7 flex items-center justify-center text-[#031354] text-[30px] rounded-md"
              style={{ borderRadius: 2 }}
            >
              <FaYoutube />
            </Link>

            {/* Trait vertical */}
            <div className="h-6 border-l-2 border-[#fff] mx-2"></div>

            {/* EN/FR */}
            <div className="text-white text-sm">
            <LanguageSelector />
            </div>
          </div>

          <div className="container flex justify-between items-center px-6 mx-auto text-white">
            <Link
              href="/"
              className="flex items-center text-[25px] font-semibold police6"
            >
              <img
                src={"/images/logo-chc.png"}
                style={{ width: 140, marginTop: -30 }}
              />
            </Link>

            {/* Mobile Button */}
            <div className="lg:hidden">
              <button
                onClick={toggleMobileMenu}
                className="text-[white] text-3xl"
              >
                <svg
                  xmlns="http://www.w3.org/2000/svg"
                  className="w-8 h-8"
                  fill="none"
                  stroke="currentColor"
                  strokeWidth="2"
                >
                  <path
                    strokeLinecap="round"
                    strokeLinejoin="round"
                    d="M4 6h16M4 12h16M4 18h16"
                  ></path>
                </svg>
              </button>
            </div>

            {/* Desktop Menu */}
            <div className="hidden lg:flex items-center gap-8">
              {navItems.map((item, index) => (
                <li
                  key={index}
                  className="group relative"
                  style={{ listStyleType: "none" }}
                >
                  <Link
                    href={item.href}
                    className="text-[#fff] group-hover:text-[#fff] transition-colors police7 text-[15px]"
                    style={{ fontWeight: 600 }}
                  >
                    {item.name}
                  </Link>
                  <div className="absolute bottom-0 left-0 w-full h-1 bg-[#fff] scale-x-0 group-hover:scale-x-100 transition-transform"></div>
                </li>
              ))}
              <Link href={"/"}>
                <FaSearch className="transform rotate-90 text-[20px]" />
              </Link>
            </div>
          </div>
        </div>

        {/* Mobile Menu */}
        <div
          ref={menuRef}
          className={`lg:hidden fixed inset-0 bg-[#1d1d1d] bg-opacity-90 transition-transform duration-300 ease-in-out ${
            isMobileMenuOpen
              ? "transform translate-x-0"
              : "transform translate-x-full"
          }`}
        >
          <button
            onClick={toggleMobileMenu}
            className="absolute top-4 right-6 text-white text-3xl"
          >
            <FaTimes />
          </button>

          <div className="flex flex-col items-center py-10 space-y-6">
            {navItems.map((item, index) => (
              <Link
                key={index}
                href={item.href}
                className="text-white text-xl py-3 hover:text-[#2e76bc] transition-colors"
                onClick={() => setIsMobileMenuOpen(false)}
              >
                {item.name}
              </Link>
            ))}
            
          </div>
        </div>
      </nav>
    </div>
  );
}
